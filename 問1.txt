# 所属部署テーブルの所属部署名
# 社員テーブルの社員名(氏+名)
# タイムカードテーブルの勤務時間を取得する
SELECT
	d.depart_name AS 所属部署名,
	CONCAT(e.l_name,e.f_name) AS 社員氏名,
	t.work_time AS 勤務時間
# 社員テーブルと所属部署テーブルの所属部署コードで内部結合する
# 社員テーブルの社員コードとタイムカードテーブルの社員コードで内部結合する
FROM
	(
		employee AS e
	INNER JOIN
		depart AS d
	ON
		e.depart_id = d.depart_id
	)
INNER JOIN
	time_card AS t
ON
	e.s_id = t.s_id
# 社員コードが「DA00001」のレコードを対象とする
# かつ、タイムカードテーブルの日付が2012/12/1～2012/12/31のレコードを対象とする
WHERE
	e.s_id='DA00001'
AND
	t.r_date BETWEEN '2012-12-01' AND '2012-12-31'
# タイムカードテーブルの日付の昇順にソートする
ORDER BY
	t.r_date
;
