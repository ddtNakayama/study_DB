# 社員名(氏)+社員名(名)、日付、勤務時間を取得する
SELECT
	CONCAT(e.l_name,e.f_name) AS 氏名,
	t.r_date,
	t.work_time
# 社員テーブルとタイムカードテーブルを一致する社員レコードで結合する
FROM
	employee AS e
INNER JOIN
	time_card AS t
ON
	e.s_id = t.s_id
# 日付が2012/12/1～2012/12/31のレコードを対象とする
WHERE
	t.r_date BETWEEN '2012-12-01' AND '2012-12-31'
# 社員コード、日付の昇順にソートする
ORDER BY
	e.s_id ASC,
	t.r_date ASC
;
