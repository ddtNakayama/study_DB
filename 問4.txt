	SELECT
		a.name_kana
	FROM
		author AS a
# 重複を含まないのであればUNIONが正しい
UNION ALL
	SELECT
		# 列数が合わないため、CONCATする必要がある
		u.l_name_kana,
		u.f_name_kana
	FROM
		usr AS u
ORDER BY
	1 DESC
;
