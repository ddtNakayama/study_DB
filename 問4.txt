SELECT
	*
FROM
	access_log
WHERE
	# 2013年1月分を対象とするならINではなくBETWEEN
	access_date IN ('2013-01-01','2013-01-31')
ORDER BY
	# リンク元のURLが昇順にソートされてしまう
	referer, ip_address DESC
;



SELECT
	*
FROM
	access_log

	access_date BETWEEN '2013-01-01' AND '2013-01-31'
ORDER BY
	referer DESC, ip_address DESC
;
